    \posterbox[adjusted title=Problem]{
        name=problem,
        column=1,
        span=\lcol,
        below=title,
    }{
        \begin{itemize}[leftmargin=*]
            \item Discrete single-impurity Anderson model $H = H_0 + H_1$ with
                  \begin{align*}
                      H_0
                       & =
                      \sum_j c_j^\dag E_j c_j
                      + \sum_{j \neq \mathrm{imp}} c_\mathrm{imp}^\dag V_j^\dag c_j + c_j^\dag V_j c_\mathrm{imp}
                      \qquad E_j, V_j \in \mathbb{C}^{N \times N} \\
                      H_1
                       & =
                      \sum_{1234 \, \in \, \mathrm{imp}} U_{1234} c_1^\dag c_2^\dag c_3 c_4
                      -  \sum_{12 \, \in \, \mathrm{imp}} U_{12}^\mathrm{MF} c_1^\dag c_2
                  \end{align*}
            \item Correlators are described by \myemph{discrete spectral representation} (DSR)
                  \begin{equation*}
                      G_0(z) = \frac{1}{z - E_\mathrm{imp} - \underbrace{\sum_j V_j^\dag \frac{1}{z - E_j} V_j}_{\scalebox{1.0}{$\Delta_0(z)$}}}
                  \end{equation*}
            \item \myemph{Dyson equation} does \myemph{not guarantee causality} numerically
            \item \myemph{Idea}: Use equation of motion to calculate DSR of self-energy
                  \begin{align*}
                      \Sigma(z) = \Sigma^\HF + \sum_j \frac{W_j''}{z - E_j''}
                  \end{align*}
                  with weights $W_j'' = (V_j'')^\dag V_j''$
            \item \myemph{Goal}:
                  Benchmark different implementations
                  on the real axis
                  at $T=0$
                  on the $d \to \infty$ Bethe lattice
        \end{itemize}
    }

    \posterbox[adjusted title=Method]{
        name=method,
        column*=2,
        span=\rcol,
        below=title,
    }{
        \begin{itemize}[leftmargin=*]
            \item Use interacting Hamiltonian $H_1$ to define auxiliary operator $q \coloneq [c_\mathrm{imp}, H_1]$. \\
                  Calculate propagator
                  \begin{equation*}
                      \tG(z)
                      =
                      \begin{pmatrix}
                          \tG_{11} & \tG_{12} \\
                          \tG_{21} & \tG_{22}
                      \end{pmatrix}
                      \coloneq
                      \begin{pmatrix}
                          \lehmann{c_\mathrm{imp}}{c_\mathrm{imp}^\dag} & \lehmann{c_\mathrm{imp}}{q^\dag} \\
                          \lehmann{q}{c_\mathrm{imp}^\dag}              & \lehmann{q}{q^\dag}
                      \end{pmatrix}
                      \qquad
                  \end{equation*}
                  with Lehmann representation
                  $\lAngle A, B^\dag \rAngle \propto \sum_{mn} \sandwich{n}{A}{m} \sandwich{m}{B^\dag}{n}$ \\
                  and static term
                  $\Sigma^\HF = \expval{\{c_\mathrm{imp},q^\dag\}} = \expval{\{q,c_\mathrm{imp}^\dag\}}$
            \item \citeauthor{Kugler2022} showed~\cite{Kugler2022}
                  \begin{equation*}
                      \Sigma(z)
                      =
                      \Sigma^\HF + \tG_{22}(z)  - \tG_{21}(z) \frac{1}{\tG_{11}(z)} \tG_{12}(z)
                  \end{equation*}
            \item \myemph{New Method}: Use Schur complement
                  \begin{equation*}
                      \left[
                          \tG^{-1}(z)
                          \right]_{22}
                      =
                      \left[
                          \begin{pmatrix}
                              \tG_{11} & \tG_{12} \\
                              \tG_{21} & \tG_{22}
                          \end{pmatrix}^{-1}
                          \right]_{22}
                      =
                      \left(
                      \tG_{22}  - \tG_{21} \frac{1}{\tG_{11} }  \tG_{12}
                      \right)^{-1}
                  \end{equation*}
                  to guarantee causal self-energy
                  \begin{equation*}
                      \Sigma(z)
                      =
                      \Sigma^\HF + \left( \bigl[\tG^{-1}(z) \bigr]_{22} \right)^{-1}
                  \end{equation*}
        \end{itemize}
    }
