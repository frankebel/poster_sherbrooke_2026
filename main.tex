    \posterbox[adjusted title=Problem]{
        name=problem,
        column=1,
        span=\lcol,
        below=title,%
    }{
        \begin{itemize}[leftmargin=*]
            \item Single-impurity Anderson model $H = H_0 + H_1$ with
                  \begin{align*}
                      H_0
                       & =
                      \sum_\sigma \epsilon_\mathrm{imp} n_{\mathrm{imp}\sigma}
                      +
                      \sum_{j\sigma} \epsilon_j n_{j\sigma}
                      + \sum_{j\sigma} V_j \left( c_{\mathrm{imp}\sigma}^\dag c_{j\sigma} + \mathrm{h. c.}\right) \, , \\
                      H_1
                       & =
                      U n_{\mathrm{imp}\uparrow}n_{\mathrm{imp}\downarrow}  \, .
                  \end{align*}
            \item Correlators are described by \myemph{discrete spectral representation} (DSR)
                  \begin{equation*}
                      G_0^{-1}(z) = z - \epsilon_\mathrm{imp} - \Delta_0(z)
                  \end{equation*}
                  with
                  \begin{equation*}
                      \Delta_0(z) = \sum_j \frac{V_j^2}{z - \epsilon_j} \, .
                  \end{equation*}
            \item \myemph{Dyson equation} does \myemph{not guarantee} $-\Im \Sigma(\omega) \ge 0$ numerically.
                  \begin{center}
                      \input{graphics/dyson-eq/dyson-eq.tex}
                  \end{center}
            \item \myemph{Idea}: Use equation of motion to calculate DSR of self-energy
                  \begin{align*}
                      \Sigma(z) = \Sigma^\HF + \sum_i \frac{w_i}{z - \epsilon_i}
                  \end{align*}
                  guaranteeing $w_i \ge 0$.
            \item \myemph{Goal}:
                  Benchmark different implementations
                  on the real axis
                  at $T=0$
                  on the $d \to \infty$ Bethe lattice.
        \end{itemize}
    }

    \posterbox[adjusted title=Method]{
        name=method,
        column*=2,
        span=\rcol,
        below=title,%
    }{
        \begin{itemize}[leftmargin=*]
            \item Define auxiliary operator $q \coloneq [c_\mathrm{imp}, H_1]$
                  and calculate propagator
                  \begin{equation*}
                      \tG(z)
                      \coloneq
                      \begin{pmatrix}
                          G_{11} & G_{12} \\
                          G_{21} & G_{22}
                      \end{pmatrix}
                      \coloneq
                      \begin{pmatrix}
                          \lehmann{c_\mathrm{imp}}{c_\mathrm{imp}^\dag} & \lehmann{c_\mathrm{imp}}{q^\dag} \\
                          \lehmann{q}{c_\mathrm{imp}^\dag}              & \lehmann{q}{q^\dag}
                      \end{pmatrix}
                      \qquad
                  \end{equation*}
                  with Lehmann representation
                  $\lAngle A, B^\dag \rAngle \propto \sum_{mn} \sandwich{n}{A}{m} \sandwich{m}{B^\dag}{n}$ \\
                  and static term
                  $\Sigma^\HF = \expval{\{c_\mathrm{imp},q^\dag\}} = \expval{\{q,c_\mathrm{imp}^\dag\}}$.
            \item \citeauthor{Kugler2022} showed~\cite{Kugler2022}
                  \begin{equation*}
                      \Sigma(z)
                      =
                      \Sigma^\HF + G_{22}  - G_{21} \frac{1}{G_{11}} G_{12} \, .
                  \end{equation*}
            \item \myemph{New Method}: Use Schur complement
                  \begin{equation*}
                      \left[
                          \tG^{-1}(z)
                          \right]_{22}
                      =
                      \left[
                          \begin{pmatrix}
                              G_{11} & G_{12} \\
                              G_{21} & G_{22}
                          \end{pmatrix}^{-1}
                          \right]_{22}
                      =
                      \left(
                      G_{22}  - G_{21} \frac{1}{G_{11} }  G_{12}
                      \right)^{-1}
                  \end{equation*}
                  to get DSR
                  \begin{equation*}
                      \Sigma(z)
                      =
                      \Sigma^\HF + \left( \bigl[\tG^{-1}(z) \bigr]_{22} \right)^{-1} \, .
                  \end{equation*}
        \end{itemize}
    }
